npx shadcn@latest add https://atlasui.vercel.app/r/blob-mouse-trailer-js.json

<TabsContent value="manual">

<Steps>

<Step>Copy and paste the following code into your project.</Step>

```html
<div class="trailer" id="trailer"></div>
```

<Step>Add `style.css`</Step>

```css
body {
  margin: 0;
  padding: 0;
  height: 100vh;
  width: 100%;
  background: #18181b;
}

.trailer {
  position: fixed;
  height: 24rem;
  aspect-ratio: 1;
  background: linear-gradient(to right, #a855f7, #f59e0b);
  border-radius: 50%;
  left: 0;
  top: 0;
  transform: translate(-50%, -50%);
  transform-origin: center;
  pointer-events: none;
  z-index: -10;
  filter: blur(100px);
  animation: rotate 10s infinite;
}

@keyframes rotate {
  from {
    transform: translate(-50%, -50%) rotate(0deg);
  }
  to {
    transform: translate(-50%, -50%) rotate(360deg);
  }
}
```

<Step>Add `script.js`</Step>

```js
const trailer = document.getElementById("trailer");

document.body.onpointermove = (e) => {
  const { clientX, clientY } = e;

  trailer.animate(
    {
      left: `${clientX}px`,
      top: `${clientY}px`,
    },
    {
      duration: 3000,
      fill: "forwards",
    },
  );
};
```

</Steps>

## Usage

<UsageTabs>

<TabsContent value="React">

```tsx
import { BlobMouseTrailer } from "@/components/blob-mouse-trailer";
```

Place the Component in your `layout.tsx` to make the Mouse Trailer accessible in entire project

```tsx
export default function RootLayout({
  children,
}: {
  children: React.ReactNode;
}) {
  return (
    <html lang="en">
      <body>
        {children}
        <BlobMouseTrailer />
      </body>
    </html>
  );
}
```

</TabsContent>

<TabsContent value="HTML">

Place the following code in your `index.html` to use the Mouse Trailer.

```html
<div class="trailer" id="trailer"></div>
```
